Git Learn:

Learn one:
//创建版本库
	git init 				//构建一个git仓库
	
	//添加文件到git仓库，一共分两步
	git add Readme.txt			//第一步：添加Readme.txt文件到git仓库
	git commit -m "添加Readme.txt文件"	//第二部：使用git commit命令，完成

Learn two:
//时光机穿梭
	git status				//git status命令，时刻掌握git仓库当前的状态
	git diff "Readme.txt"			//查看文件具体修改了哪些内容

	//提交修改和提交更新一样分两步
	git add Readme.txt			//第一步：添加Readme.txt文件到git仓库
	git commit -m "添加Readme.txt文件"	//第二部：使用git commit命令，完成
